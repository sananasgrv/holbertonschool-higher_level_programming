"""
Doctest for MyList class
"""

>>> MyList = __import__('1-my_list').MyList

# Test case #0: Disordered list of positive integers
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

# Test case #1: Reverse sorted list of positive integers
>>> my_list = MyList()
>>> my_list.append(5)
>>> my_list.append(4)
>>> my_list.append(3)
>>> my_list.append(2)
>>> my_list.append(1)
>>> print(my_list)
[5, 4, 3, 2, 1]
>>> my_list.print_sorted()
